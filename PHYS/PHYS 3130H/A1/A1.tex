% PROBLEM 1
\begin{problem}
A Skier weighing $\qty{90}{\kilo\gram}$ starts from rest down a hill inclined at $\qty{17}{\degree}$.
He skis $\qty{100}{\meter}$ down the hill and then coasts for $\qty{70}{\meter}$ along level snow until he stops.
Find the coefficient of kinetic friction between the skis and the snow. What is his velocity when he reaches the bottom of the hill?
\end{problem}
\begin{soln}
  On the hill we can use conservation of energy to find the velocity the skier will have at the bottom of the slope.
  At the top of the hill the skier has potential energy
  $$U=mgh=100mg\sin(17)$$
  and at the bottom all of this will have been converted into kinetic energy or lost to friction as work,
  $$\frac{1}{2}mv^2=mgh-100\mu_kmg\implies v=\sqrt{2g(h-100\mu_k)}$$
  where we've taken the positive root implicitly because the skier should not have negative velocity.
  Now we have sufficient initial conditions to investigate the flat part of the problem, (by shifting our origin) $v(t=0)=v(x=0)=\sqrt{2g(h-100\mu_k)}$ and $x(t=0)=0$.
  Our equation of motion in this region is
  $$F=m\ddot{x}=\mu_kN=-mg\mu_k\implies \frac{d^2x}{dt^2}=-g\mu_k$$
  which we rewrite using $\frac{d^2x}{dt^2}=\frac{dv}{dt}=\frac{dv}{dx}\frac{dx}{dt}=v\frac{dv}{dx}$ as
  $$v\frac{dv}{dx}=-g\mu_k\implies v^2/2=-g\mu_kx+C_1\implies v=\sqrt{C_1-2g\mu_kx}.$$
  applying $v(x=0)=\sqrt{2g(h-100\mu_k)}$ here,
  $$v(0)=\sqrt{C_1}=\sqrt{2g(h-100\mu_k)}\implies C_1=2g(h-100\mu_k)\implies v(x)=\sqrt{2g(h-100\mu_k-\mu_kx)}.$$
  Now we know that $v(70)=0$ so
  $$v(70)=\sqrt{2g(h-100\mu_k-70\mu_k)}=0\implies \mu_k=\frac{h}{170}$$
  so
  $$\frac{h}{170}=\frac{100\sin(17)}{170}\approx 0.17$$
  now we just substitute this into our initial condition to determine that the skiers velocity at the bottom of the hill is
  $$v(x=0)=\sqrt{2g(h-100\frac{100\sin(17)}{170})}=\sqrt{2g(100\sin(17)-\frac{10000\sin(17)}{170})}\approx \qty{15.4}{\meter\per\second}$$
\end{soln}

\begin{problem}~
\begin{enumerate}[label=(\alph*)]
  \item If a projectile is fired from the origin of the coordinate system with an initial velocity $v_0$ and
        in a direction making an angle $\alpha$ with the horizontal, calculate the time required for the projectile
        to cross a line passing through the origin and making an angle $\beta < \alpha$ with the
        horizontal.
  \item When a projectile is fired with an initial
        speed of $v_0$, it passes through the points $P$ and
        $Q$ which are at a distance $h$. Find the distance
        between $P$ and $Q$ in the figure
\end{enumerate}
\begin{center}
  \begin{tikzpicture}[scale=1.2]
    \fill[gray!20] (-1,-1) rectangle (6,0);
    \draw[thick] (-1,0) -- (6,0);
    \draw[<->, thick] (2.45,0.05) -- (2.45,1.25-0.05) node[midway, right]{$h$};


    \draw[thick, ->, domain=0:4.9, smooth, variable=\t]
    plot[samples=100] ({\t}, {\t*(1-(1/5)*\t)});

    \draw[->, thick, rotate=45] (0,0) -- (1.3,0) node[above left] {$v_0$};

    \draw (0.4,0) arc[start angle=0, end angle=45, radius=0.4,dashed];
    \draw[draw=none, rotate=22.5] (0,0) -- (0.6,0) node {$\alpha$};
    \draw[dotted, thick] (0,0) -- (0.7,0);
  \end{tikzpicture}
\end{center}
\end{problem}
\begin{soln}~
  \begin{enumerate}[label=(\alph*)]
    \item To do this we'll need to find the point of intersection between $y_p(x)$, the trajectory of the projectile, and $y_l(x)=\tan(\beta)x$, the given line.
          To do this we solve the equations of motion for both $y(t)$ and $x(t)$. Starting with $x$,
          \begin{align*}
            F_x      & =m\ddot{x}=0 \\
            \implies & \dot{x}=C    \\
            \implies & x(t)=Ct+C_1
          \end{align*}
          and because $x(0)=0$ and $\dot{x}(0)=v_{0x}$, $C_1=0$ and $C=v_{0x}$ so $x(t)=v_{0x}t$. Now for $y(t)$,
          \begin{align*}
            F_y      & =m\ddot{y}=-mg            \\
            \implies & \dot{y}=-gt+C             \\
            \implies & y=-\frac{gt^2}{2}+Ct+C_1.
          \end{align*}
          Applying initial conditions again we get that $y(t)=-\frac{gt^2}{2}+v_{0y}t$. We then solve for $t(x)=\frac{x}{v_{0x}}$ and substitute this into $y(t)$ to obtain
          $$y(x)=-\frac{gx^2}{2v_{0x}^2}+\frac{v_{0y}x}{v_{0x}}.$$
          Now we just solve
          $$\tan(\beta)x=-\frac{g}{2v_{0x}^2}x^2+\frac{v_{0y}}{v_{0x}}x\implies \frac{2v_{0x}}{g}\left[v_{0x}\tan(\beta)-v_{0y}\right]=x$$
          Hence, using $x(t)=v_{0x}t$ we get that 
          $$\frac{2v_{0x}}{g}\left[v_{0x}\tan(\beta)-v_{0y}\right]=v_{0x}t\implies t=
          \frac{2}{g}\left[v_0\cos\beta\tan(\beta)-v_0\sin\beta\right]=\frac{2v_0}{g}\left[\cos(\alpha)\tan(\beta)-\sin(\alpha)\right]$$
    \item To do this we just shift down the trajectory by $h$ so that $P$ and $Q$ are zeroes, so
          \begin{align*}
            0           & =-\frac{g}{2v_{0x}^2}x^2+\frac{v_{0y}}{v_{0x}}x-h                                                                                                                 \\
            \implies  x & =\frac{-\frac{v_{0y}}{v_{0x}}\pm\sqrt{\left(\frac{v_{0y}}{v_{0x}}\right)^2-4\left(-\frac{g}{2v_{0x}^2}\right)\left(h\right)}}{2\left(-\frac{g}{2v_{0x}^2}\right)} \\
                        & =\frac{v_{0y}v_{0x}\mp v_{0x}\sqrt{2gh+v_{0y}^2}}{g}
          \end{align*}
          then we can take the difference $Q-P$. With $Q$ being the greater,
          $$\frac{v_{0y}v_{0x}+ v_{0x}\sqrt{2gh+v_{0y}^2}}{g}-\frac{v_{0y}v_{0x}- v_{0x}\sqrt{2gh+v_{0y}^2}}{g}=\frac{2v_0\cos(\alpha)\sqrt{2gh+v_0^2\sin^2(\alpha)}}{g}$$

  \end{enumerate}
\end{soln}

\begin{problem}
A particle is under the influence of a force $F=-kx+\frac{k}{\alpha^2}x^3$, where $k$ and $\alpha$
are constants and $k$ is positive.
\begin{enumerate}[label=(\alph*)]
  \item Determine $U(x)$ and plot $U(x)$ as a function of $x$ using

\end{enumerate}
\end{problem}
\begin{soln}~

    \item $$U(x)=-\int F\,dx=-\int -kx+\frac{k}{\alpha^2}x^3\,dx=\frac{k}{2}x^2-\frac{k}{4\alpha^2}x^4$$

            \end{tikzpicture}
          \end{center}
    \item The points at $E_1$ are unstable equilibrium points in the potential, i.e. zeroes in the force. Hence we solve

  \end{enumerate}
\end{soln}

% PROBLEM 4
\begin{problem}
A block of mass $m=\qty{1.62}{\kilo\gram}$ slides down a frictionless incline
which has a loop at the end as shown in the figure. The block is
released at a height $h=\qty{3.91}{\meter}$ above the bottom of the loop.
\begin{enumerate}[label=(\alph*)]
  \item What is the force of the inclined track on the block at the
        bottom (Point A)?
  \item What is the force of the track on the block at point B?
        And hence at what speed does the block leave the track?
\end{enumerate}
\end{problem}
\begin{soln}~
  \begin{enumerate}[label=(\alph*)]
    \item At point A the block is in circular motion, hence the net force must be the centripetal force
          $$F_{net}=F_{C}=\frac{mv_A^2}{R}$$
          The only force actually acting on the block is the force of gravity pointing straight down so
          $$F_{net}=F_{N}-mg=\frac{mv_A^2}{R}\implies F_{N}=m\left(\frac{v_A^2}{R}+g\right).$$
          By conservation of energy we know that at this point all the potential energy that the block had at the beginning of the track,
          $mgh$, has been converted into kinetic energy, so
          $$mgh=\frac{1}{2}mv_A^2\implies 2gh=v_A^2$$
          which we can substitute back into our equation for $F_N$ to obtain
          $$F_{N}=m\left(\frac{v_A^2}{R}+g\right)=mg\left(\frac{2h}{R}+1\right)$$
    \item Again we make the argument that as the block is in circular motion, the net force must be $F_C=\frac{mv_B^2}{R}$. Here however the
          reaction force of the ramp due to gravity is lessend due to the slope. Hence $F_g=mg\cos\theta$ so
          $$F_{net}=F_C=F_N-F_g\implies \frac{mv_B^2}{R}=F_N-mg\cos\theta\implies m\left(\frac{v_B^2}{R}+g\cos(\pi/4)\right)=F_N.$$
          Now we must determine $v_B$. This can be done with conservation of energy. By conservation of energy then
          $$\frac{1}{2}mv_A^2=\frac{1}{2}mv_B^2+mg\pr{h}\implies v_B^2=v_A^2-2g\pr{h},$$
          i.e. the kinetic energy at A is equal to the kinetic energy at B plus the little bit of loss to the height in the form of gravitational potential.
          To find $\pr{h}$ we use some geometry. We know that the right triangle formed by
          the line of length $R$ intersecting $A$ has hypotenuse $H$ such that $H\cos\theta=R$. $H$ is $R$ plus a little bit that we get from sweeping out the circle.
          This little bit is the hypotenuse of a smaller (similar) triangle whose upright portion is $\pr{h}$ and whose hypotenuse is $K$. From sine law we know that
          this smaller triangle obeys $$\frac{\pr{h}}{\sin(\pi/4)}=K$$ so
          $$H=R+\frac{\pr{h}}{\sin(\pi/4)}$$ and hence
          $$R=\cos(\pi/4)\left[R+\frac{\pr{h}}{\sin(\pi/4)}\right]=R\cos(\pi/4)+\pr{h}\implies \pr{h}=R\left(1-\frac{1}{\sqrt{2}}\right).$$
          Putting this back into our expression for the velocity we obtain
          $$v_B^2=v_A^2-2gR\left(1-\frac{1}{\sqrt{2}}\right)=2gh-2gR\left(1-\frac{1}{\sqrt{2}}\right)=2g\left[h-R\left(1-\frac{1}{\sqrt{2}}\right)\right]$$
          which we put into our expression for the normal force to obtain
          $$F_N=m\left(\frac{1}{R}2g\left[h-R\left(1-\frac{1}{\sqrt{2}}\right)\right]+g\cos(\pi/4)\right)=mg\left(\frac{2h}{R}-2+\frac{3}{\sqrt{2}}\right)$$
          which I don't believe can be evaluated further as I can't figure out a way to determine $R$.
  \end{enumerate}
\end{soln}

% PROBLEM 5
\begin{problem}
Determine if the following forces are conservative. If so find $U(\bv{r})$
\begin{enumerate}[label=(\alph*)]
  \item $\bv{F}=(ayz+bx+c)\ux+(axz+bz)\uy+(axy+by)\uz$; $a,b,c$ are constants.
  \item $\bv{F}=\dfrac{a\uv{r}}{r}$
\end{enumerate}
\end{problem}
\begin{soln}Assuming we are in a simply connected domain (i.e. real physical space) then it is sufficient to show that $\nabla\cross\bv{F}=0$ to imply a
  conservative force field. So,
  \begin{enumerate}[label=(\alph*)]
    \item $$\curl{\bv{F}}=
            \begin{vmatrix}
              \ux        & \uy        & \uz        \\
              \partial_x & \partial_y & \partial_z \\
              ayz+bx+c   & axz+bz     & axy+by
            \end{vmatrix}=[(ax+b)-(ax+b)]\ux-[(ay)-(ay)]\uy+[(az)-(az)]\uz=\bv{0}
          $$
          and 
          $$U=-\int F\implies U(\bv{r})=-(axyz+\frac{bx^2}{2}+cx+K)\ux-(axyz+byz+Q)\uy-(axyz+byz+T)\uz$$
          where $K,Q,T$ are constants of integration.
    \item Here I will use the formula given in the front cover of Griffiths' Electrodynamics for curl in spherical coordinates. This formula is quite long. Thankfully here
    we only have a dependence on $r$ so other partials will be zero. This means we obtain 
    $$\curl{\bv{F}}=-\frac{1}{r}\partial_r(rF_r)\uv{\theta}$$
    which is just the derivative of a constant ($a$) thanks to the multiplication by $r$, so this field is indeed conservative. The potential is 
    $$U(\bv{r})=U(r)=-a\ln(r)+C$$
    where we can directly integrate in spherical without fear as the gradient does not feature any multiplications inside of derivative arguments which might cause us trouble.
  \end{enumerate}
\end{soln}
