% PROBLEM 1
\begin{problem} Find all ideals of the ring $\mathbb{Z}_{60}$. Identify the prime ideals and the
maximal ideals (justify).
\end{problem}
\begin{soln}
  We defined an ideal to be an additive subgroup. Since $(\mathbb{Z}_{60},+)$ is cyclic we know that there is a subgroup for each
  divisor of $60$. These divisors are $1$,
  $2$,
  $3$,
  $4$,
  $5$,
  $6$,
  $10$,
  $12$,
  $15$,
  $20$,
  $30$, and
  $60$. The subgroups $\hull{1}$ and $\hull{60}$ are obviously ideals, the rest must be checked.
  Recall that an additive subgroup is an ideal $I$ if (in the commutative case) for $i\in I$ and $r\in R$,
  $ri\in I$. This is clearly the case for every divisor we found as $ri$ will just be $kn$ for $k$ a divisor
  and $n\in \mathbb{Z}$, $0\leq n < 60$ and $kn$ is hence an integer multiple of the divisor and will therefore belong to the
  additive subgroup $\hull{k}$ which is the ideal. I will check the maximal ideals first as they will also be prime. Recall that a maximal
  ideal of a ring $R$ is an ideal which is not properly contained in any other ideal of $R$. Borrowing from normal subgroups,
  for generators $a$ and $b$, $\hull{a}\subset \hull{b}$ iff $a\in \hull{b}$. Clearly then the only maximal ideals will be those with prime
  generators because $\hull{p}$ will only contain things like $np$ for $n\in \mathbb{Z}$. The only prime-generated ideals we found were
  $\hull{2}$, $\hull{3}$, and $\hull{5}$ and so these are the only maximal ideals. We get for free the previously mentioned maximal ideals
  as also being prime ideals. We can quickly show that others are not prime as
  \begin{enumerate}[label=(\roman*)]
    \item $2\cdot2=4\in\hull{4}$, $2\notin \hull{4}$
    \item $3\cdot 2=6\in \hull{6}$, $2\notin \hull{6}$
    \item $5\cdot 2=10\in \hull{10}$, $2\notin \hull{10}$
    \item $2\cdot 6=12\in\hull{12}$, $2\notin \hull{12}$
    \item $3\cdot 5=15\in\hull{15}$, $3\notin \hull{15}$
    \item $2\cdot 10=20\in\hull{20}$, $2\notin \hull{20}$
    \item $2\cdot 15=30\in\hull{30}$, $2\notin \hull{30}$
  \end{enumerate}
  however, $\hull{60}$ is a prime ideal as it contains only $\left\{0\right\}$ and the only way to get zero is to multiply by some multiple of
  $60$, all of which are ``in'' $\hull{60}$ when under mod 60.
\end{soln}

% PROBLEM 2
\begin{problem} Let
$I = \left\{(3x, y)\mid x,y \in\mathbb{Z}\right\}$
\begin{enumerate}[label=(\alph*)]
  \item Show that $I$ is a maximal ideal of the ring $\mathbb{Z}\times \mathbb{Z}$.
  \item What if $3x$ is replaced with $4x$?
  \item Can we make a general statement about this?
\end{enumerate}
\end{problem}
\begin{soln}~
  \begin{enumerate}[label=(\alph*)]
    \item $\mathbb{Z}\times \mathbb{Z}/I$ is the set of remainders of element-wise division of tuples. Breaking this down across the direct product
          this looks like $\mathbb{Z}/\left\{3x\mid x\in\mathbb{Z}\right\}=\mathbb{Z}/3\mathbb{Z}$ and $\mathbb{Z}/\left\{y\in\mathbb{Z}\right\}=\mathbb{Z}/\mathbb{Z}$
          which is just $\mathbb{Z}_3\times \left\{0\right\}\cong \mathbb{Z}_3$. Since $p$ is prime, $\mathbb{Z}_3$ is a field which implies the quotienting
          set $I$ is maximal.
    \item Using the same tactic, $\mathbb{Z}\times \mathbb{Z}/\left\{(4x, y)\mid x,y\in\mathbb{Z}\right\}\cong \mathbb{Z}_4\times \left\{0\right\}\cong \mathbb{Z}_4$
          which is not a field as $4$ is not prime and hence the ideal $\left\{(4x, y)\mid x,y\in\mathbb{Z}\right\}$ is not maximal.
    \item Yes, ideals of the form $\left\{(px, y)\mid x,y \in\mathbb{Z}\right\}$ for prime $p\in\mathbb{Z}$ are maximal in
          $\mathbb{Z}\times \mathbb{Z}$. This is because you end up with the $\mathbb{Z}_p$ structure which is a field and hence the ideal was maximal.
  \end{enumerate}
\end{soln}

% PROBLEM 4
\begin{problem}
Show by example that the intersection of two prime ideals need not
be a prime ideal.
\end{problem}
\begin{soln}
  The ring $\mathbb{Z}$ has prime ideals $\hull{2}$ and $\hull{3}$ since
  $\hull{2}=2\mathbb{Z}$ and clearly any $ab\in 2\mathbb{Z}$ requires that either
  $a$ or $b$ is a multiple of two and is hence in $2\mathbb{Z}$. The same can be argued for
  $\hull{3}$. Their intersection is all things that are multiples of 3 \emph{and} 2, i.e.
  multiples of $2\cdot 3=6$ so $\hull{6}=\hull{2}\cap \hull{3}$. $\hull{6}$ is not prime as
  $2\cdot3=6\in \hull{6}$ but neither $3$ nor $2$ are in $\hull{6}$.
\end{soln}

\begin{problem}
Let $R$ be a principal ideal domain and $I$ an ideal of $R$. Show that
every ideal of $R/I$ is principal.
\end{problem}
\begin{soln}
  I'm not happy with this but it's the best I have:
  We know that the map $\phi: R\to R/I$ given by $\phi(a)=a+I$ is a homomorphism. We also know that
  (in the commutative case) and ideal of $R$ is a set where for $r\in R$ and $i \in I$, $ri\in I$. Consider an ideal
  $J$ of $R$. Because $J$ is an ideal we have that for $j\in J$ and $r\in R$ $rj\in J\implies \phi(rj)\in \phi\left[J\right]$
  so $\phi(r)\phi(j)\in \phi[J]$. This shows that $J+I$, i.e. $\left\{j+I\mid j\in J\right\}=J/I$ is an ideal of $R/I$ because
  $\phi(r)\phi(j)\in \phi[J]$ satisfies the definition of an ideal for elements of the form $\phi(j)=j+I$ and $\phi(r)\in R/I$.
  Since we chose $J$ as an ideal of $R$ it is principal as $R$ is a PID and since $J$ is arbitrary, all ideals of $R/I$ are
  principal.
\end{soln}

% PROBLEM 6
\begin{problem}
Let $R = \left\{a+bi\mid a, b \in\mathbb{Z}\right\}$. Show that $R/\hull{1-i}$ is a field, and determine
its number of elements.
\end{problem}
\begin{soln}
  In $R/\hull{1-i}$ we can treat $i$ as $x$, an indeterminate. Then this becomes a polynomial ring quotiented by a polynomial ideal.
  Specifically it is all integer coefficient polynomials of degree at most 1 quotiented by the ideal generated by a polynomial of degree 1. Since here,
  with $I=\hull{1-i}$,
  $1-i+I=I\equiv 0$ and so, working ``mod $1-i$'', $1-i\equiv 0\implies 1\equiv i$ however we define $i$ to be the number where $i^2=-1$
  so $1^2\equiv-1\implies 2\equiv 0$. At this point this feels like $\mathbb{Z}_2$ but let's check. For a
  general $a+bi\in R$ we've determined that, mod $1-i$, $a+bi\equiv a+b$. And we've shown that
  if $a+b$ contains a factor of $2$, $a+b\equiv 0$. If $a+b$ does not contain a factor of $2$ we've shown that $a+b\equiv 1$.
  This is in fact $\mathbb{Z}_2$ which we know to be a field with two elements.
\end{soln}
