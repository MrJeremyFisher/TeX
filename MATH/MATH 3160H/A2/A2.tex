% PROBLEM 1
\begin{problem}
A simple pendulum initially has the length $\ell_0$ and makes an angle $\theta_0$ with the vertical line. It is
then released from this position. If the length of the pendulum increases with time $t$ according to
$\ell=\ell_0+\epsilon t$, where $\epsilon$ is a small constant, then the angle $\theta$ that the pendulum makes with the vertical
line, assuming that the oscillations are small, is the solution of the initial value problem
$$\left(\ell_0+\epsilon t\right)\frac{d^2\theta}{dt^2}+2\epsilon\frac{d\theta}{dt}+g\theta=0,\quad \theta(0)=\theta_0,\quad \frac{d\theta}{dt}(0)=0$$
where $g$ is the acceleration due to gravity. Find the solution of the above initial value problem.
\end{problem}
\begin{soln}
  Consider the change of variables
  $$\ell-\ell_0=\epsilon t.$$ This gives us
  $d\ell=\epsilon dt$
  and so
  $$\frac{d\theta}{dt}=\epsilon\frac{d\theta}{d\ell}$$ and
  $$\epsilon^2\frac{d^2\theta}{d\ell^2}$$
  so the ODE becomes
  $$\ell \epsilon^2\frac{d^2\theta}{d\ell^2}+2\epsilon^2\frac{d\theta}{d\ell}+g\theta=0$$
  which we can divide through by $\epsilon^2$ to see that
  $$\ell\frac{d^2\theta}{d\ell^2}+2\frac{d\theta}{d\ell}+\frac{g}{\epsilon^2}\theta=0$$
  which is very nearly the Bessel equation, we just need to fiddle a bit more. We can multiply by $\ell$ to get
  $$\ell^2\frac{d^2\theta}{d\ell^2}+2\ell\frac{d\theta}{d\ell}+\frac{g\ell}{\epsilon^2}\theta=0$$
  which is of a form we know,
  $$x^2y''+(1-2a)xy'+\left[b^2c^2x^{2c}+(a^2-p^2c^2)\right]y=0$$
  which has solution
  $$y=c_1x^aJ_p(bx^c)+c_2x^aY_p(bx^c).$$
  Here then
  \begin{align*}
    1-2a     & =2            \\
    b^2c^2   & =g/\epsilon^2 \\
    2c       & =1            \\
    a^2-pc^2 & =0
  \end{align*}
  which tells us that
  \begin{align*}
    c & =1/2                \\
    a & =-1/2               \\
    b & =2\sqrt{g}/\epsilon \\
    p & =1
  \end{align*}
  so we get a solution of the form
  $$\theta(\ell)=c_1\ell^{-1/2}J_1\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)+c_2\ell^{-1/2}Y_1\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right).$$
  Now recall we originally had initial conditions in $t$ so we need to change those to $\ell$.
  Since $\ell=\ell_0+\epsilon t$, when $t=0$ $\ell=\ell_0$ so $\theta(t=0)=\theta_0\implies \theta(\ell=\ell_0)=\theta_0$
  and $\frac{d\theta}{dt}(t=0)=0\implies \frac{d\theta}{d\ell}(\ell=\ell_0)=0$.
  So we take the derivative of our solution,
  \begin{align*}
    \frac{d\theta(\ell)}{d\ell} & =\frac{d}{d\ell}\left[
      c_1\ell^{-1/2}J_1\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)
      +c_2\ell^{-1/2}Y_1\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)
    \right]                                                     \\
                                & =-\frac{1}{2\ell^{3/2}}\left[
      c_1J_1\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)
      +c_2Y_1\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)
    \right]                                                     \\
                                & \quad+
    \frac{\sqrt{g}}{2\epsilon\ell}\left[
      c_1\left[J_0\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)-J_2\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)\right]
      +c_2\left[Y_0\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)-Y_2\left(\frac{2\sqrt{g}}{\epsilon}\ell^{1/2}\right)\right]
    \right]                                                     \\
  \end{align*}
  and evaluate the condition,
  \begin{align*}
    \theta_\ell(\ell_0)=0
             & =-\frac{1}{2\ell_0^{3/2}}\left[
      c_1J_1\left(\frac{2\sqrt{g\ell_0}}{\epsilon}\right)
      +c_2Y_1\left(\frac{2\sqrt{g}}{\epsilon}\right)
    \right]                                    \\
             & \quad+
    \frac{\sqrt{g}}{2\epsilon\ell_0}\left[
      c_1\left[J_0\left(\frac{2\sqrt{g\ell_0}}{\epsilon}\right)-J_2\left(\frac{2\sqrt{g\ell_0}}{\epsilon}\right)\right]
      +c_2\left[Y_0\left(\frac{2\sqrt{g\ell_0}}{\epsilon}\right)-Y_2\left(\frac{2\sqrt{g\ell_0}}{\epsilon}\right)\right]
    \right]                                    \\
    \implies &
    c_2\left\{
    +\frac{1}{2\ell_0^{3/2}}Y_1\left(\frac{2\sqrt{g}}{\epsilon}\right)
    -\frac{\sqrt{g}}{2\epsilon\ell_0}\left[Y_0\left(\alpha\right)-Y_2\left(\alpha\right)\right]
    \right\}
    =
    c_1\left\{
    \frac{\sqrt{g}}{2\epsilon\ell_0}\left[J_0\left(\alpha\right)-J_2\left(\alpha\right)\right]
    -\frac{1}{2\ell_0^{3/2}}J_1\left(\alpha\right)
    \right\}
  \end{align*}
  for $\alpha=\frac{2\sqrt{g\ell_0}}{\epsilon}$
  Now we apply the condition $\theta(\ell=\ell_0)=\theta_0$,
  \begin{align*}
    \theta(\ell=\ell_0)=\theta_0 & =\frac{1}{\sqrt{\ell_0}}\left[c_1J_1\left(\alpha\right)+c_2Y_1\left(\alpha\right)\right] \\
    \implies                     & \frac{\sqrt{\ell_0}\theta_0-c_1J_1\left(\alpha\right)}{Y_1\left(\alpha\right)}=c_2       \\
  \end{align*}
\end{soln}

\begin{problem}
The differential equation
$$x^2\frac{d^2E}{dx^2}-\mu x\frac{dE}{dx}-k^2x^\nu E=0$$
occurs in the study of the flow of current through electrical transmission lines; here $\mu$, $\nu$ are positive
constants and $E$ represents the potential difference (with respect to one end of the line) at a point a
distance $x$ from that end of the line. Find its general solution.
\end{problem}
\begin{soln}
  Recall we found that the ODE
  $$x^2y''+(1-2a)xy'+\left[b^2c^2x^{2c}+(a^2-p^2c^2)\right]y=0$$
  has solution
  $$y=c_1x^aJ_p(bx^c)+c_2x^aY_p(bx^c).$$
  The given problem is of this form with come of the coefficients being complex. We have that
  \begin{align*}
    1-2a       & =-\mu \\
    b^2c^2     & =-k^2 \\
    2c         & =\nu  \\
    a^2-p^2c^2 & =0
  \end{align*}
  which gives
  \begin{align*}
    a & =\frac{1}{2}(\mu+1)  \\
    b & =2i\frac{k}{\nu}     \\
    c & =\frac{\nu}{2}       \\
    p & =\frac{(\mu+1)}{\nu}
  \end{align*}
  so we get a solution of the form
  $$E=x^{\frac{1}{2}(\mu+1)}\left[c_1J_{\frac{(\mu+1)}{\nu}}(2i\frac{k}{\nu} x^{\frac{1}{2}\nu})+c_2Y_{\frac{(\mu+1)}{\nu}}(2i\frac{k}{\nu} x^{\frac{1}{2}\nu})\right].$$
\end{soln}

\begin{problem} Use the definition of the Bessel functions to prove each of the following.
\begin{enumerate}[label=(\alph*)]
  \item $\displaystyle J_{-1/2}(x)=\sqrt{\frac{2}{\pi x}}\cos x$
  \item $\displaystyle J_{3/2}(x)=\sqrt{\frac{2}{\pi x}}\left(x^{-1}\sin x-\cos x\right)$
  \item $\displaystyle J_{-3/2}(x)=\sqrt{\frac{2}{\pi x}}\left(-x^{-1}\cos x-\sin x\right)$
\end{enumerate}
\end{problem}
\begin{soln} Recall that the series definition of the Bessel function is
  $$J_n(x) = \sum_{m=0}^\infty \frac{(-1)^m}{m!\Gamma(m+n+1)} {\left(\frac{x}{2}\right)}^{2m + n}$$
  \begin{enumerate}[label=(\alph*)]
    \item From the series definition
          \begin{align*}
            J_{-1/2}(x) & = \sum_{m=0}^\infty \frac{(-1)^m}{m!\Gamma(m+1/2)} {\left(\frac{x}{2}\right)}^{2m - 1/2}             \\
                        & = \sqrt{\frac{2}{x}}\sum_{m=0}^\infty \frac{(-1)^m}{m!\Gamma(m+1/2)} {\left(\frac{x}{2}\right)}^{2m}
          \end{align*}
          and $\Gamma(m+1/2)=\Gamma(m-1/2)(m-1/2)=\Gamma(m-3/2)(m-1/2)(m-3/2)\dots =\frac{\sqrt{\pi}(2m-1)(2m-3)\dots 3\cdot 1}{2^m}$ so
          \begin{align*}
            J_{-1/2}(x) & = \sqrt{\frac{2}{x}}\sum_{m=0}^\infty \frac{(-1)^m}{m!\Gamma(m+1/2)} {\left(\frac{x}{2}\right)}^{2m}                                 \\
                        & = \sqrt{\frac{2}{\pi x}}\sum_{m=0}^\infty \frac{(-1)^m}{m! (2m-1)(2m-3)\dots 3\cdot 1\cdot 2^m} x^{2m}                               \\
                        & = \sqrt{\frac{2}{\pi x}}\sum_{m=0}^\infty \frac{(-1)^m}{(m)(m-1)(m-2)\dots 3\cdot2\cdot1 (2m-1)(2m-3)\dots 3\cdot 1\cdot 2^m} x^{2m} \\
                        & = \sqrt{\frac{2}{\pi x}}\sum_{m=0}^\infty \frac{(-1)^m}{(2m)(2m-2)(2m-4)\dots 6\cdot 4\cdot 2 (2m-1)(2m-3)\dots 3\cdot 1} x^{2m}     \\
                        & = \sqrt{\frac{2}{\pi x}}\sum_{m=0}^\infty \frac{(-1)^m}{(2m)!} x^{2m}                                                                \\
                        & =\sqrt{\frac{2}{\pi x}} \cos x.
          \end{align*}
    \item From the series definition
          \begin{align*}
            J_{3/2}(x) & = \sum_{m=0}^\infty \frac{(-1)^m}{m!\Gamma(m+5/2)} {\left(\frac{x}{2}\right)}^{2m + 3/2}                                        \\
                       & = \frac{x^{3/2}}{2^{3/2}}\sum_{m=0}^\infty \frac{(-1)^m}{m!\Gamma(m+5/2)} {\left(\frac{x}{2}\right)}^{2m}                       \\
                       & = \frac{x^{3/2}}{2^{3/2}}\sum_{m=0}^\infty \frac{(-1)^m }{m!2^m\sqrt{\pi} (2m+3)(2m+1)(2m-1)(2m-3)\dots 5\cdot 3\cdot 1} x^{2m} \\
                       & = \frac{1}{2}\sqrt{\frac{x}{2\pi}}\sum_{m=0}^\infty \frac{(-1)^m }{(2m+1)! (2m+3)} x^{2m+1}                                   \\                        
          \end{align*}
    \item $\displaystyle J_{-3/2}(x)=\sqrt{\frac{2}{\pi x}}\left(-x^{-1}\cos x-\sin x\right)$
  \end{enumerate}
\end{soln}
